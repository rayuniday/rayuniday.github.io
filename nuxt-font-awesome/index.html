    Nuxt에 Font Awesome 사용하기

    yarn add -D
    @nuxtjs/fontawesome
    @fortawesome/free-brands-svg-icons
    @fortawesome/free-regular-svg-icons
    @fortawesome/free-solid-svg-icons

    nuxt.config.js

    const solid = ["faBars"];
const regular = [];
const brands = [];

export default {
  buildModules: [
    ["@nuxtjs/fontawesome", { component: "fontAwesome", suffix: true }],
  ],
  fontawesome: {
    icons: {
      solid: solid,
      regular: regular,
      brands: brands,
    },
  },
};

    <font-awesome-icon icon="bars" />
